### 1) Problem: Most Stones Removed with Same Row or Column

**Given stones placed on a 2D grid, a stone can be removed if another stone exists in the same row or column. The goal is to determine the maximum number of stones that can be removed while still leaving at least one stone per connected component.**

**Approach:**

* Treat each row and column as nodes and connect them using Disjoint Set Union (Union-Find).
* For every stone `(x, y)`, union the row node `x` with the column node `n + y + 1`.
* Track all nodes that actually appear and count how many unique connected components exist.
* Maximum removable stones = `total stones - number of connected components`.

**Time Complexity:** `O(N * α(N))`
**Space Complexity:** `O(N + maxRow + maxCol)`

**Code:**

```cpp
class DisjointSet{
    private:
        vector<int> parent,size;
    public:
        DisjointSet(int n){
            size.resize(n,1);
            parent.resize(n);
            for(int i=0; i<n; i++) parent[i]=i;
        }
        int findUPar(int node){
            if(parent[node]==node) return node;
            return parent[node]=findUPar(parent[node]);
        }
        void unionBySize(int u, int v){
            int ulp_u = findUPar(u);
            int ulp_v = findUPar(v);
            if(ulp_u == ulp_v) return;
            if(size[ulp_u]<size[ulp_v]){
                parent[ulp_u]=ulp_v;
                size[ulp_v]+=size[ulp_u];
            } else {
                parent[ulp_v]=ulp_u;
                size[ulp_u]+=size[ulp_v];
            }
        }     
};
class Solution {
public:
    int removeStones(vector<vector<int>>& stones) {
        int s = stones.size();
        int n=-1,m=-1;
        for(auto it: stones){
            n=max(n,it[0]);
            m=max(m,it[1]);
        }
        set<int> st;
        DisjointSet ds(n+m+2);
        for(auto it: stones){
            ds.unionBySize(it[0],n+it[1]+1);
            st.insert(it[0]);
            st.insert(n+1+it[1]);
        }
        int k=0;
        for(auto it: st){
            if(ds.findUPar(it)==it) k++;
        }
        return s-k;
    }
};
```
