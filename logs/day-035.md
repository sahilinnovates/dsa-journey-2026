### 1) Problem: Making a Large Island

Given an n x n binary grid, you can change at most one 0 to 1. An island is formed by connected 1s (4-directional). Determine the maximum possible island size after performing this operation.

**Approach:**

* Use Disjoint Set (Union-Find) to group all existing connected 1s into components.
* Convert each cell to a unique node using index = r * n + c.
* For every 0 cell, check its 4-directional neighbors and collect unique island components.
* Sum the sizes of these unique components and add 1 for the flipped cell.
* Track the maximum island size obtained.

**Time Complexity:** `O(N^2)`
**Space Complexity:** `O(N^2)`

**Code:**

```cpp
class DisjointSet{
    private:
        vector<int> parent,size;
    public:
        DisjointSet(int n){
            size.resize(n,1);
            parent.resize(n);
            for(int i=0; i<n; i++) parent[i]=i;
        }
        int findUPar(int node){
            if(parent[node]==node) return node;
            return parent[node]=findUPar(parent[node]);
        }
        void unionBySize(int u, int v){
            int ulp_u = findUPar(u);
            int ulp_v = findUPar(v);
            if(ulp_u==ulp_v) return;
            if(size[ulp_u]<size[ulp_v]){
                parent[ulp_u]=ulp_v;
                size[ulp_v]+=size[ulp_u];
            } else {
                parent[ulp_v]=ulp_u;
                size[ulp_u]+=size[ulp_v];
            }
        }
        int maxCompSize(){
            return *max_element(size.begin(),size.end());
        }
        int getSize(int node){
            return size[node];
        }
};
class Solution {
public:
    int largestIsland(vector<vector<int>>& grid) {
        int n = grid.size();
        DisjointSet ds(n*n);
        queue<pair<int,int>> q;
        vector<int> row = {-1,0,1,0};
        vector<int> col = {0,1,0,-1};
        for(int r=0; r<n; r++){
            for(int c=0; c<n; c++){
                if(grid[r][c]==1){
                    for(int i=0; i<4; i++){
                        int nr = r+row[i];
                        int nc = c+col[i];
                        if(nr>=0 && nc>=0 && nr<n && nc<n && grid[nr][nc]==1){
                            ds.unionBySize(r*n+c,nr*n+nc);
                        }
                    }
                } else {
                    q.push({r,c});
                }
            }
        }
        int maxs = ds.maxCompSize();
        while(!q.empty()){
            int r = q.front().first;
            int c = q.front().second;
            q.pop();
            set<int> st;
            for(int i=0; i<4; i++){
                int nr = r+row[i];
                int nc = c+col[i];
                if(nr>=0 && nc>=0 && nr<n && nc<n && grid[nr][nc]==1){
                    st.insert(ds.findUPar(nr*n+nc));
                }
            }
            int sum=1;
            for(auto it : st){
                sum+=ds.getSize(it);
            }
            maxs = max(maxs,sum);
        }
        return maxs;
    }
};
```
