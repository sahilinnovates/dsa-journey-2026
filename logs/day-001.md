### 1) Problem: Number of Enclaves

**You can move outside the grid only by stepping on land cells (1) in 4 directions. Return the count of land cells from where you can’t go outside.**
**Approach:**

* Push all boundary land cells into a queue and run BFS.
* Mark all land cells reachable from the boundary as visited (these can escape).
* Count land cells that remain unvisited → these are enclaves.

**Time Complexity:** `O(n*m)`
**Space Complexity:** `O(n*m)`

**Code:**

```cpp
int numberOfEnclaves(vector<vector<int>> &grid) {
        int n = grid.size();
        int m = grid[0].size();
        queue<pair<int,int>> q;
        vector<vector<int>> vis(n,vector<int>(m,0));
        for(int i=0; i<m; i++){
            if(grid[0][i]==1 && vis[0][i]==0){
                q.push({0,i});
                vis[0][i]=1;
            }
            if(grid[n-1][i]==1 && vis[n-1][i]==0){
                q.push({n-1,i});
                vis[n-1][i]=1;
            }
        }
        for(int i=0; i<n; i++){
            if(grid[i][0]==1 && vis[i][0]==0){
                q.push({i,0});
                vis[i][0]=1;
            }
            if(grid[i][m-1]==1 && vis[i][m-1]==0){
                q.push({i,m-1});
                vis[i][m-1]=1;
            }
        }
        int row[]={-1,0,1,0};
        int col[]={0,1,0,-1};
        while(!q.empty()){
            int r = q.front().first;
            int c = q.front().second;
            q.pop();
            for(int i=0; i<4; i++){
                int nr=r+row[i];
                int nc=c+col[i];
                if(nr>=0 && nc>=0 && nr<n && nc<m && grid[nr][nc]==1 && vis[nr][nc]==0){
                    q.push({nr,nc});
                    vis[nr][nc]=1;
                }
            }
        }
        int cnt=0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(grid[i][j]==1 && vis[i][j]==0){
                    cnt++;
                }
            }
        }
        return cnt;
    }
```

### 2) Problem: Rotten Oranges

**2 = rotten orange, 1 = fresh orange, 0 = empty cell. Every minute, fresh oranges adjacent (4-dir) to rotten become rotten. Return min minutes to rot all fresh oranges, else -1.**
**Approach:**

* Push all initially rotten oranges into a queue with time = 0 (multi-source BFS).
* For each rotten orange, rot its adjacent fresh oranges and push them with time + 1.
* Track fresh count; if any fresh remains at the end → return `-1`, else return max time.

**Time Complexity:** `O(n*m)`
**Space Complexity:** `O(n*m)`

**Code:**

```cpp
int orangesRotting(vector<vector<int>> &grid) {
        vector<vector<int>> res = grid;
        int n = grid.size();
        int m = grid[0].size();
        int fresh=0;
        queue<pair<pair<int,int>,int>> q;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(res[i][j]==2){
                    q.push({{i,j},0});
                } else if(res[i][j]==1){
                    fresh++;
                }
            }
        }
        if(fresh==0) return 0;
        int maxt=0;
        int row[]={-1,0,1,0};
        int col[]={0,1,0,-1};
        while(!q.empty()){
            int t=q.front().second;
            int r=q.front().first.first;
            int c= q.front().first.second;
            q.pop();
            maxt=max(maxt,t);
            for(int i=0; i<4; i++){
                int nr=r+row[i];
                int nc=c+col[i];
                if(nr>=0 && nc>=0 && nr<n && nc<m && res[nr][nc]==1){
                    q.push({{nr,nc},t+1});
                    res[nr][nc]=2;
                    fresh--;
                }
            }
        }
        if(fresh!=0) return -1;
        return maxt;
    }
```
