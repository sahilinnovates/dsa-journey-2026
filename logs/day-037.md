### 1) Problem: Kosaraju's Algorithm

**Given a directed graph represented using an adjacency list, the task is to find the number of strongly connected components (SCCs) in the graph. A strongly connected component is a group of vertices where every vertex is reachable from every other vertex in the same component.**

**Approach:**

* Perform a DFS traversal and push nodes into a stack based on their finishing time.
* Reverse all the edges of the graph to form the transpose graph.
* Process nodes in the order of the stack and run DFS on the transpose graph.
* Each DFS traversal in this phase identifies one strongly connected component.

**Time Complexity:** `O(V + E)`
**Space Complexity:** `O(V + E)`

**Code:**

```cpp
class Solution{
public:
    void dfs1(vector<int> adj[], vector<int>& vis, stack<int>& st, int v){
        vis[v]=1;
        for(int x: adj[v]){
            if(!vis[x]) dfs1(adj,vis,st,x);
        }
        st.push(v);
    }
    void dfs2(vector<int> adjList[], vector<int>& vis, int v){
        vis[v]=1;
        for(int x: adjList[v]){
            if(!vis[x]) dfs2(adjList,vis,x);
        }
    }
    int kosaraju(int V, vector<int> adj[]){
        stack<int> st;
        vector<int> vis(V,0);
        for(int i=0; i<V; i++){
            if(!vis[i]) dfs1(adj,vis,st,i);
        }
        vector<int> adjList[V];
        for(int i=0; i<V; i++){
            for(int x: adj[i]){
                adjList[x].push_back(i);
            }
        }
        int cnt=0;
        fill(vis.begin(),vis.end(),0);
        while(!st.empty()){
            int v = st.top();
            st.pop();
            if(!vis[v]){
                cnt++;
                dfs2(adjList,vis,v);
            }
        }
        return cnt;
    }
};
```
