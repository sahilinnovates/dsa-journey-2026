### 1) Problem: Find MST Weight using Kruskal's Algorithm

**short description of problem**
Given a weighted undirected connected graph represented using an adjacency list, compute the total weight of the Minimum Spanning Tree (MST) using Kruskal’s algorithm.

**Approach:**

* Extract all edges from the adjacency list into a list with (weight, u, v) format.
* Sort all edges in ascending order of weight.
* Use Disjoint Set (Union-Find) with union by size and path compression.
* Add an edge to MST only if it connects two different components.

**Time Complexity:** `O(E log E)`
**Space Complexity:** `O(V + E)`

**Code:**

```cpp
class DisjointSet {
private:
    vector<int> size,parent;
public:
    DisjointSet(int n) {
        parent.resize(n+1);
        size.resize(n+1,1);
        for(int i=0; i<=n; i++) parent[i]=i;
    }

    int findUPar(int node){
        if(node==parent[node]) return node;
        return parent[node]=findUPar(parent[node]);
    }

    bool find(int u, int v) {
        return (findUPar(u)==findUPar(v));
    }

    void unionBySize(int u, int v) {
        int ulp_u = findUPar(u);
        int ulp_v = findUPar(v);
        if(ulp_u==ulp_v) return;
        if(size[ulp_u]<size[ulp_v]){
            parent[ulp_u]=ulp_v;
            size[ulp_v]+=size[ulp_u];
        } else {
            parent[ulp_v]=ulp_u;
            size[ulp_u]+=size[ulp_v];
        }
    }
};
class Solution{
    public:
    int spanningTree(int V, vector<vector<int>> adj[]) {
        vector<pair<int,pair<int,int>>> edges;
        for(int i=0; i<V; i++){
            for(auto it: adj[i]){
                edges.push_back({it[1],{i,it[0]}});
            }
        }
        DisjointSet ds(V);
        sort(edges.begin(),edges.end());
        int sum=0;
        for(auto it: edges){
            int wt = it.first;
            int u = it.second.first;
            int v = it.second.second;
            if(ds.findUPar(u) != ds.findUPar(v)){
                sum+=wt;
                ds.unionBySize(u,v);
            }
        }
        return sum;
    }
};
```
