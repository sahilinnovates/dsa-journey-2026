### 1) Problem: Accounts Merge

Given a list of accounts where each account contains a name followed by email addresses, merge accounts that belong to the same person. Two accounts belong to the same person if they share at least one common email.

**Approach:**

* Use Disjoint Set (Union-Find) where each account index represents a node.
* Map each email to the first account index where it appears.
* If an email appears again, union the two account indices.
* After unions, group emails by their ultimate parent account.
* Sort the emails and attach the account name to build the final merged list.

**Time Complexity:** `O(N * M log M)`
**Space Complexity:** `O(N * M)`

**Code:**

```cpp
class DisjointSet{
    private:
        vector<int> size,parent;
    public:
        DisjointSet(int n){
            parent.resize(n);
            size.resize(n,1);
            for(int i=0; i<n; i++) parent[i]=i;
        }
        int findUPar(int node){
            if(node==parent[node]) return node;
            return parent[node]=findUPar(parent[node]);
        }
        bool find(int u,int v){
            return (findUPar(u)==findUPar(v));
        }
        void unionBySize(int u,int v){
            int ulp_u = findUPar(u);
            int ulp_v = findUPar(v);
            if(ulp_u==ulp_v) return;
            if(size[ulp_u]<size[ulp_v]){
                parent[ulp_u]=ulp_v;
                size[ulp_v]+=size[ulp_u];
            } else {
                parent[ulp_v]=ulp_u;
                size[ulp_u]+=size[ulp_v];
            }
        }
};
class Solution {
public:
    vector<vector<string>> accountsMerge(vector<vector<string>>& accounts) {
        int n = accounts.size();
        DisjointSet ds(n);
        unordered_map<string,int> mpp;
        for(int i=0; i<n ; i++){
            for(int j=1,s=accounts[i].size(); j<s; j++){
                if(mpp.find(accounts[i][j])==mpp.end()){
                    mpp.insert({accounts[i][j],i});
                } else {
                    ds.unionBySize(mpp[accounts[i][j]],i);
                }
            }
        }
        vector<vector<string>> ans;
        vector<string> mails[n];
        for(auto it: mpp){
            mails[ds.findUPar(it.second)].push_back(it.first);
        }
        for(int i=0; i<n; i++){
            if(mails[i].size()==0) continue;
            sort(mails[i].begin(),mails[i].end());
            vector<string> tmp;
            tmp.push_back(accounts[i][0]);
            for(auto s: mails[i]){
                tmp.push_back(s);
            }
            ans.push_back(tmp);
        }
        return ans;
    }
};
```
