### 1) Problem: Number of Islands II

**short description of problem**
Given an initially empty grid of size n × m (all water), land is added at specific coordinates through a sequence of operations. After each operation, determine the number of islands present in the grid. Islands are connected horizontally or vertically.

**Approach:**

* Use Disjoint Set (Union-Find) to dynamically manage connected land components.
* Convert each cell into a unique node using index = r * m + c.
* When land is added, increase island count and check its 4-directional neighbors.
* If neighboring land belongs to a different component, union them and reduce island count.
* Store the island count after each operation.

**Time Complexity:** `O(K * α(N*M))`
**Space Complexity:** `O(N*M)`

**Code:**

```cpp
class DisjointSet{
    private:
        vector<int> size,parent;
    public:
        DisjointSet(int n){
            parent.resize(n);
            size.resize(n,1);
            for(int i=0; i<n; i++) parent[i]=i;
        }
        int findUPar(int node){
            if(node==parent[node]) return node;
            return parent[node]=findUPar(parent[node]);
        }
        bool find(int u,int v){
            return (findUPar(u)==findUPar(v));
        }
        void unionBySize(int u,int v){
            int ulp_u = findUPar(u);
            int ulp_v = findUPar(v);
            if(ulp_u==ulp_v) return;
            if(size[ulp_u]<size[ulp_v]){
                parent[ulp_u]=ulp_v;
                size[ulp_v]+=size[ulp_u];
            } else {
                parent[ulp_v]=ulp_u;
                size[ulp_u]+=size[ulp_v];
            }
        }
};
class Solution{
public:
    vector<int> numOfIslands(int n, int m, vector<vector<int>> &A){
        vector<int> ans;
        int s = A.size();
        DisjointSet ds(n*m);
        vector<vector<int>> mat (n, vector<int>(m,0));
        int cnt=0;
        vector<int> row = {-1,0,1,0};
        vector<int> col = {0,1,0,-1};
        for(int i=0; i<s; i++){
            int r = A[i][0];
            int c = A[i][1];
            //
            if(mat[r][c]==1){
                ans.push_back(cnt);
                continue;
            }
            mat[r][c]=1;
            cnt++;
            int node = A[i][0]*m+A[i][1];
            for(int j=0; j<4; j++){
                int nr = r+row[j];
                int nc = c+col[j];
                if(nr>=0 && nc>=0 && nr<n && nc<m && mat[nr][nc]==1){
                    if(!(ds.find(r*m+c,nr*m+nc))){
                        ds.unionBySize(r*m+c,nr*m+nc);
                        cnt--;
                    }
                }
            }
            ans.push_back(cnt);
        }
        return ans;
    }
};
```
